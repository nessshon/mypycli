# i18n Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Добавить строгий механизм локализации в `mypycli` — ленивый `Translator` с eager-load одного языка, префиксированный библиотечный каталог в `mypycli:`, CLI `mypycli locales init|sync|check` для скаффолдинга пользовательского каталога.

**Architecture:** `Translator(locales_dir)` сканирует `*.yml` при init, `set_language` читает только выбранный файл, nested YAML сплющивается в dotted-ключи. `Application` принимает Translator-инстанс, резолвит язык из `db.language`/`LANG`/`en` и пробрасывает в приватный `mypycli.i18n.internal` для библиотечного кода. CLI-тулинг работает через `importlib.resources`.

**Tech Stack:** Python 3.10+, `pyyaml`, `importlib.resources`, pytest.

from __future__ import annotations

import re

_VALID_LANG_RE = re.compile(r"^[a-z]{2}$")


def parse_lang_env(raw: str) -> str:
    """Parse env LANG-style string to a 2-letter lowercase code, or empty string."""
    if not raw:
        return ""
    head = raw.split(".", 1)[0].split("_", 1)[0].lower()
    return head if _VALID_LANG_RE.match(head) else ""
```

```python
from __future__ import annotations

from pathlib import Path
from typing import Any

import yaml


class FlattenError(ValueError):
    pass


def flatten(
    data: dict[str, Any],
    *,
    existing: dict[str, str] | None = None,
    prefix: str = "",
) -> dict[str, str]:
    """Flatten nested dict to dotted-key flat dict. Values must be str."""
    out: dict[str, str] = dict(existing or {})
    _walk(data, prefix, out)
    return out


def _walk(node: Any, prefix: str, out: dict[str, str]) -> None:
    if not isinstance(node, dict):
        raise FlattenError(f"Expected dict at '{prefix or '<root>'}', got {type(node).__name__}")
    for key, value in node.items():
        path = f"{prefix}.{key}" if prefix else str(key)
        if isinstance(value, dict):
            if path in out:
                raise FlattenError(f"Collision at '{path}': leaf and branch conflict")
            _walk(value, path, out)
        elif isinstance(value, str):
            if any(k.startswith(f"{path}.") for k in out):
                raise FlattenError(f"Collision at '{path}': leaf and branch conflict")
            out[path] = value
        else:
            raise FlattenError(f"Non-string value at '{path}': {type(value).__name__}")


def load_flat(path: Path) -> dict[str, str]:
    """Parse YAML file and return flattened dotted-key dict."""
    try:
        raw = yaml.safe_load(path.read_text(encoding="utf-8"))
    except yaml.YAMLError as exc:
        raise FlattenError(f"Invalid YAML in {path}: {exc}") from exc
    if raw is None:
        return {}
    if not isinstance(raw, dict):
        raise FlattenError(f"Root of {path} must be a mapping, got {type(raw).__name__}")
    return flatten(raw)
```

from __future__ import annotations

from pathlib import Path

from mypycli.i18n.loader import load_flat


class Translator:
    def __init__(self, locales_dir: Path | str) -> None:
        self._dir = Path(locales_dir)
        if not self._dir.is_dir():
            raise FileNotFoundError(f"Locales directory not found: {self._dir}")
        self._available: set[str] = {p.stem for p in self._dir.glob("*.yml")}
        self._catalog: dict[str, str] | None = None
        self._language: str | None = None

    def available_languages(self) -> set[str]:
        return set(self._available)

    @property
    def language(self) -> str | None:
        return self._language
```

    def set_language(self, lang: str) -> None:
        if lang not in self._available:
            raise ValueError(
                f"Language '{lang}' not available. Available: {sorted(self._available)}"
            )
        self._catalog = load_flat(self._dir / f"{lang}.yml")
        self._language = lang

    def __call__(self, key: str, **kwargs: object) -> str:
        if self._catalog is None:
            raise RuntimeError("Translator not initialized — call set_language() first")
        tmpl = self._catalog.get(key)
        if tmpl is None:
            raise LookupError(
                f"Missing translation: '{key}' for lang '{self._language}'"
            )
        return tmpl.format(**kwargs) if kwargs else tmpl
```

from __future__ import annotations

from importlib.resources import files
from pathlib import Path

import yaml


def _library_locales() -> dict[str, dict]:
    """Return {lang: parsed_yaml} from bundled mypycli/locales/."""
    root = files("mypycli").joinpath("locales")
    result: dict[str, dict] = {}
    for entry in root.iterdir():
        if entry.is_file() and entry.name.endswith(".yml"):
            lang = entry.name[:-4]
            raw = entry.read_text(encoding="utf-8")
            parsed = yaml.safe_load(raw) or {}
            result[lang] = parsed
    return result


def cmd_init() -> int:
    target = Path.cwd() / "locales"
    target.mkdir(exist_ok=True)

    library = _library_locales()
    if not library:
        print("No bundled locales found in mypycli package.")
        return 1

    for lang, data in sorted(library.items()):
        out = target / f"{lang}.yml"
        if out.exists():
            print(f"skip  locales/{lang}.yml (exists)")
            continue
        mypycli_section = {"mypycli": data.get("mypycli", {})}
        out.write_text(
            yaml.safe_dump(
                mypycli_section, allow_unicode=True, sort_keys=False
            ),
            encoding="utf-8",
        )
        print(f"✓ locales/{lang}.yml created")
    return 0
```

def cmd_check() -> int:
    target = Path.cwd() / "locales"
    if not target.is_dir():
        print("locales/ directory not found")
        return 1

    library = _library_locales()
    user_files = sorted(target.glob("*.yml"))
    if not user_files:
        print("No locale files found in locales/")
        return 1

    has_error = False

    # 1) mypycli: consistency per file vs library
    for file in user_files:
        lang = file.stem
        if lang not in library:
            print(f"info  locales/{lang}.yml: no library counterpart, skipping mypycli check")
            continue
        user_data = yaml.safe_load(file.read_text(encoding="utf-8")) or {}
        user_mypy = _flat_keys(user_data.get("mypycli") or {})
        lib_mypy = _flat_keys(library[lang].get("mypycli") or {})
        if user_mypy != lib_mypy:
            missing = lib_mypy - user_mypy
            extra = user_mypy - lib_mypy
            print(f"✗ locales/{lang}.yml: mypycli section out of sync (run 'mypycli locales sync')")
            if missing:
                print(f"  missing: {sorted(missing)}")
            if extra:
                print(f"  extra:   {sorted(extra)}")
            has_error = True

    # 2) User keys identical across languages
    per_lang_user: dict[str, set[str]] = {}
    for file in user_files:
        data = yaml.safe_load(file.read_text(encoding="utf-8")) or {}
        user_data = {k: v for k, v in data.items() if k != "mypycli"}
        per_lang_user[file.stem] = _flat_keys(user_data)

    all_keys: set[str] = set().union(*per_lang_user.values()) if per_lang_user else set()
    for lang, keys in per_lang_user.items():
        missing = all_keys - keys
        if missing:
            print(f"✗ locales/{lang}.yml: missing user keys: {sorted(missing)}")
            has_error = True

    if not has_error:
        print("✓ locales consistent")
    return 1 if has_error else 0


def _flat_keys(data: dict) -> set[str]:
    out: set[str] = set()
    _walk_keys(data, "", out)
    return out


def _walk_keys(node, prefix: str, out: set[str]) -> None:
    if not isinstance(node, dict):
        return
    for k, v in node.items():
        path = f"{prefix}.{k}" if prefix else str(k)
        if isinstance(v, dict):

Проверить: `_exit_not_root` вызывается из `run()` после `build_parser(app).parse_args()`. К этому моменту `start()` НЕ вызывался (он внутри каждой `run_*` команды). Значит i18n-internal ещё не привязан → `_()` бросит `RuntimeError`.

**Решение:** в `Application.__init__` сразу после `self.translator = translator` добавить:

```python
        # Eager-bind for early CLI messages (before start()).
        # Requires catalog to be loaded — use fallback "en" if available.
        if self.translator.language is None:
            available = self.translator.available_languages()
            if "en" in available:
                self.translator.set_language("en")
                _i18n_internal.bind(self.translator)
```

Это даст fallback-английский для сообщений `_exit_not_root` и `_print_cancelled` **до** `start()`. При `start()` язык потом переключится на правильный.
